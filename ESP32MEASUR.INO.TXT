uint32_t g_capture_index = 0;

bool g_apply_contrast = true;

// Square crop settings

// -------------------------------------------------------

// -------------------------------------------------------
bool g_calibrated = false;
uint8_t g_bg_mean[CALIB_GRID][CALIB_GRID] = {};
uint8_t g_bg_stddev[CALIB_GRID][CALIB_GRID] = {};
float g_k_sigma = 2.5f; // threshold = mean - k*stddev

// FIX #13: Exposure lock after calibration
bool g_exposure_locked = false;

const framesize_t res_values[] = {FRAMESIZE_96X96, FRAMESIZE_QQVGA, FRAMESIZE_QCIF, FRAMESIZE_HQVGA, FRAMESIZE_240X240, FRAMESIZE_QVGA, FRAMESIZE_CIF, FRAMESIZE_HVGA, FRAMESIZE_VGA, FRAMESIZE_SVGA, FRAMESIZE_XGA, FRAMESIZE_HD, FRAMESIZE_SXGA, FRAMESIZE_UXGA};

bool initCamera(framesize_t frame_size = FRAMESIZE_VGA)
{
    Serial.print("Initializing camera... ");

    camera_config_t config;
    config.ledc_channel = LEDC_CHANNEL_0;
    config.ledc_timer = LEDC_TIMER_0;
    config.pin_d0 = Y2_GPIO_NUM;
    config.pin_d1 = Y3_GPIO_NUM;
    config.pin_d2 = Y4_GPIO_NUM;
    config.pin_d3 = Y5_GPIO_NUM;
    config.pin_d4 = Y6_GPIO_NUM;
    config.pin_d5 = Y7_GPIO_NUM;
    config.pin_d6 = Y8_GPIO_NUM;
    config.pin_d7 = Y9_GPIO_NUM;
    config.pin_xclk = XCLK_GPIO_NUM;
    config.pin_pclk = PCLK_GPIO_NUM;
    config.pin_vsync = VSYNC_GPIO_NUM;
    config.pin_href = HREF_GPIO_NUM;
    config.pin_sccb_sda = SIOD_GPIO_NUM;
    config.pin_sccb_scl = SIOC_GPIO_NUM;
    config.pin_pwdn = PWDN_GPIO_NUM;
    config.pin_reset = RESET_GPIO_NUM;
    config.xclk_freq_hz = 20000000;
    config.pixel_format = PIXFORMAT_GRAYSCALE;
    config.frame_size = frame_size;
    config.grab_mode = CAMERA_GRAB_WHEN_EMPTY;
    config.fb_location = CAMERA_FB_IN_PSRAM;
    config.jpeg_quality = 12;
    config.fb_count = 1;

    if (psramFound())
    {
        config.jpeg_quality = 10;
        config.fb_count = 2;
        config.grab_mode = CAMERA_GRAB_LATEST;
        Serial.println("PSRAM found");
    }
    else
    {
        if (frame_size > FRAMESIZE_SVGA)
            config.frame_size = FRAMESIZE_SVGA;
        config.fb_location = CAMERA_FB_IN_DRAM;
        Serial.println("PSRAM not found, using DRAM");
    }

    esp_err_t err = esp_camera_init(&config);
    if (err != ESP_OK)
    {
        Serial.printf("FAIL (0x%x)\n", err);
        return false;
    }

    sensor_t *s = esp_camera_sensor_get();
    if (s)
    {
#if defined(CAMERA_MODEL_AI_THINKER)
        s->set_vflip(s, 0);
        s->set_hmirror(s, 0);
#elif defined(CAMERA_MODEL_M5STACK_WIDE) || defined(CAMERA_MODEL_M5STACK_ESP32CAM)
        s->set_vflip(s, 1);
        s->set_hmirror(s, 1);
#elif defined(CAMERA_MODEL_ESP32S3_EYE)
        s->set_vflip(s, 1);
#endif
        s->set_brightness(s, 0);
        s->set_contrast(s, 1);
        s->set_saturation(s, 0);
        s->set_sharpness(s, 1);
        s->set_denoise(s, 1);
        s->set_gainceiling(s, GAINCEILING_2X);
        // Start with AEC on; will be locked after calibration
        s->set_exposure_ctrl(s, 1);
        s->set_aec2(s, 1);
        s->set_ae_level(s, 0);
        s->set_gain_ctrl(s, 1);
        s->set_agc_gain(s, 0);
        s->set_wb_mode(s, 1);
        s->set_framesize(s, frame_size);
    }

    Serial.println("OK");
    delay(100);
    return true;
}

bool initSD()
{
    Serial.print("Initializing SD card... ");
    pinMode(FLASH_LED_PIN, OUTPUT);
    digitalWrite(FLASH_LED_PIN, LOW);
    if (!SD_MMC.begin("/sdcard", true))
    {
        Serial.println("FAIL");
        return false;
    }
    Serial.println("OK");
    return true;
}

// =======================================================
// FIX prev analysis: IMPROVED CALIBRATION
//  - 7 samples, median across samples (outlier resistant)
//  - per-cell mean AND stddev stored
//  - exposure locked after calibration
//  - 4x4 grid (CALIB_GRID defined in config.h)
// =======================================================
void performCalibration()
{
    Serial.println("\n--- Calibration Start ---");
    Serial.println("REMOVE ALL OBJECTS. Ensure clean white background.");

    pinMode(FLASH_LED_PIN, OUTPUT);
    digitalWrite(FLASH_LED_PIN, LOW);

    // FIX #13: unlock exposure first so AEC can re-settle on current lighting
    sensor_t *s = esp_camera_sensor_get();
    if (s)
    {
        s->set_exposure_ctrl(s, 1);
        s->set_gain_ctrl(s, 1);
    }

    // FIX #9: discard warm-up frames before calibration samples
    Serial.println("Waiting for AEC to settle (10 warm-up frames)...");
    for (int i = 0; i < 10; i++)
    {
        camera_fb_t *fb = esp_camera_fb_get();
        if (fb)
            esp_camera_fb_return(fb);
        delay(60);
    }
    Serial.println("AEC settled.");

    // FIX prev analysis: use 7 samples for robust median
    const int NUM_SAMPLES = 7;

    // Accumulate per-cell pixel lists across all samples for statistics
    // For each cell: store per-sample mean so we can take median
    float sample_means[CALIB_GRID][CALIB_GRID][NUM_SAMPLES];
    float sample_vars[CALIB_GRID][CALIB_GRID][NUM_SAMPLES];
    int good_samples = 0;

    for (int s_idx = 0; s_idx < NUM_SAMPLES; s_idx++)
    {
        Serial.printf("\nCalibration sample %d/%d...\n", s_idx + 1, NUM_SAMPLES);

        camera_fb_t *fb = esp_camera_fb_get();
        if (!fb)
        {
            Serial.println("  Capture failed, skipping.");
            continue;
        }

        Serial.printf("  Image: %dx%d\n", fb->width, fb->height);

        // Apply crop if enabled
        int work_w = fb->width, work_h = fb->height;
        uint8_t *work_img = fb->buf;
        uint8_t *cropped = NULL;

        if (g_use_crop && g_crop_size < fb->width && g_crop_size < fb->height)
        {
            int crop_x = (fb->width - g_crop_size) / 2;
            int crop_y = (fb->height - g_crop_size) / 2;
            work_w = work_h = g_crop_size;
            cropped = (uint8_t *)ps_malloc(g_crop_size * g_crop_size);
            if (cropped)
            {
                for (int y = 0; y < g_crop_size; y++)
                    memcpy(&cropped[y * g_crop_size],
                           &fb->buf[(crop_y + y) * fb->width + crop_x],
                           g_crop_size);
                work_img = cropped;
            }
        }

        // Compute per-cell mean and variance
        int region_w = work_w / CALIB_GRID;
        int region_h = work_h / CALIB_GRID;

        for (int gy = 0; gy < CALIB_GRID; gy++)
        {
            int y0 = gy * region_h;
            int y1 = (gy == CALIB_GRID - 1) ? work_h : y0 + region_h;

            for (int gx = 0; gx < CALIB_GRID; gx++)
            {
                int x0 = gx * region_w;
                int x1 = (gx == CALIB_GRID - 1) ? work_w : x0 + region_w;

                uint64_t sum = 0;
                int count = 0;
                for (int y = y0; y < y1; y++)
                    for (int x = x0; x < x1; x++)
                    {
                        sum += work_img[y * work_w + x];
                        count++;
                    }

                float mean = (float)sum / count;

                // Variance pass
                float var = 0;
                for (int y = y0; y < y1; y++)
                    for (int x = x0; x < x1; x++)
                    {
                        float d = work_img[y * work_w + x] - mean;
                        var += d * d;
                    }
                var /= count;

                sample_means[gy][gx][good_samples] = mean;
                sample_vars[gy][gx][good_samples] = var;
            }
        }

        if (cropped)
            free(cropped);
        esp_camera_fb_return(fb);
        good_samples++;
        delay(150);
        yield();
    }

    if (good_samples == 0)
    {
        Serial.println("\nERROR: No successful calibration samples!");
        return;
    }

    Serial.printf("\nUsing %d good samples, computing median...\n", good_samples);

    // FIX prev analysis: median of sample_means[] for each cell (outlier-resistant)
    // Simple insertion sort on the small sample array
    for (int gy = 0; gy < CALIB_GRID; gy++)
    {
        for (int gx = 0; gx < CALIB_GRID; gx++)
        {
            // Sort means
            float m[NUM_SAMPLES];
            float v[NUM_SAMPLES];
            for (int i = 0; i < good_samples; i++)
            {
                m[i] = sample_means[gy][gx][i];
                v[i] = sample_vars[gy][gx][i];
            }
            for (int i = 1; i < good_samples; i++)
            {
                float km = m[i], kv = v[i];
                int j = i - 1;
                while (j >= 0 && m[j] > km)
                {
                    m[j + 1] = m[j];
                    v[j + 1] = v[j];
                    j--;
                }
                m[j + 1] = km;
                v[j + 1] = kv;
            }

            // Median
            float med_mean = m[good_samples / 2];
            float med_var = v[good_samples / 2];
            float std_dev = sqrtf(med_var);

            g_bg_mean[gy][gx] = (uint8_t)constrain((int)med_mean, 0, 255);
            g_bg_stddev[gy][gx] = (uint8_t)constrain((int)std_dev, 0, 255);
        }
    }

    g_calibrated = true;

    // FIX #13: Lock exposure now so capture uses same brightness
    s = esp_camera_sensor_get();
    if (s)
    {
        // Read current settled values
        g_locked_aec_value = s->status.aec_value;
        g_locked_agc_gain = s->status.agc_gain;

        // Lock them
        s->set_exposure_ctrl(s, 0);
        s->set_aec_value(s, g_locked_aec_value);
        s->set_gain_ctrl(s, 0);
        s->set_agc_gain(s, g_locked_agc_gain);

        g_exposure_locked = true;
        Serial.printf("Exposure locked: AEC=%d, gain=%d\n", g_locked_aec_value, g_locked_agc_gain);
    }

    Serial.println("\n=== CALIBRATION RESULTS ===");
    Serial.printf("K-sigma: %.2f\n", g_k_sigma);
    for (int gy = 0; gy < CALIB_GRID; gy++)
    {
        for (int gx = 0; gx < CALIB_GRID; gx++)
        {
            uint8_t T = (uint8_t)constrain(
                (int)(g_bg_mean[gy][gx] - g_k_sigma * g_bg_stddev[gy][gx]), 0, 255);
            Serial.printf("  [%d,%d] mean=%3d std=%3d -> T=%3d\n",
                          gx, gy, g_bg_mean[gy][gx], g_bg_stddev[gy][gx], T);
        }
    }
    Serial.println("--- Calibration Complete ---\n");
}

// =======================================================
// COMMAND PROCESSING
// =======================================================
void processCommand(String cmd)
{
    cmd.trim();
    String cmdUpper = cmd;
    cmdUpper.toUpperCase();

    if (cmdUpper == "CAPTURE")
    {
        captureAndMeasure();
    }
    else if (cmdUpper == "CALIBRATE")
    {
        performCalibration();
    }
    else if (cmdUpper == "STATUS" || cmdUpper == "HELP")
    {
        printSystemInfo();
    }
    else if (cmdUpper == "UNLOCKEXP")
    {
        // Allow user to re-enable AEC (e.g. when lighting changes)
        sensor_t *s = esp_camera_sensor_get();
        if (s)
        {
            s->set_exposure_ctrl(s, 1);
            s->set_gain_ctrl(s, 1);
        }
        g_exposure_locked = false;
        g_calibrated = false; // force recalibration
        Serial.println("Exposure unlocked. Please CALIBRATE again.");
    }
    else if (cmdUpper.startsWith("RES"))
    {
        int res_num = cmdUpper.substring(4).toInt();
        if (res_num >= 0 && res_num < res_counts)
        {
            Serial.printf("Changing to %s...\n", res_names[res_num]);
            esp_camera_deinit();
            delay(100);
            if (initCamera(res_values[res_num]))
            {
                current_res_index = res_num; // FIX #3: store index
                current_resolution = res_values[res_num];
                g_calibrated = false;
                g_exposure_locked = false;
                // FIX #10: warn user mm/px is no longer valid
                if (g_use_mm)
                    Serial.println("WARNING: Resolution changed — mm/px scale is now invalid. Recalibrate and set MM.");
                Serial.println("OK. Recalibrate before measuring.");
            }
            else
            {
                Serial.println("Failed");
                initCamera(current_resolution);
            }
        }
        else
            Serial.printf("Invalid resolution index. Range: 0-%d\n", res_counts - 1);
    }
    else if (cmdUpper.startsWith("CROP"))
    {
        String param = cmdUpper.substring(5);
        param.trim();
        if (param == "OFF")
        {
            g_use_crop = false;
            g_calibrated = false;
            Serial.println("Crop disabled. Recalibrate.");
        }
        else
        {
            int size = param.toInt();
            if (size > 0 && size <= 2000)
            {
                g_use_crop = true;
                g_crop_size = size;
                g_calibrated = false;
                Serial.printf("Crop enabled: %dx%d center. Recalibrate.\n", size, size);
            }
            else
                Serial.println("Usage: CROP <size> or CROP OFF");
        }
    }
    else if (cmdUpper.startsWith("KSIGMA"))
    {
        float k = cmd.substring(7).toFloat();
        if (k > 0.5f && k < 10.0f)
        {
            g_k_sigma = k;
            Serial.printf("K-sigma set to %.2f (recalibrate not required)\n", k);
        }
        else
            Serial.println("Usage: KSIGMA <0.5..10.0>  (e.g. KSIGMA 2.5)");
    }
    else if (cmdUpper.startsWith("OFFSET"))
    {
        g_thresh_offset = cmdUpper.substring(7).toInt();
        Serial.printf("Fallback threshold offset: %d\n", g_thresh_offset);
    }
    else if (cmdUpper.startsWith("CONTRAST"))
    {
        g_apply_contrast = (cmdUpper.substring(9).toInt() == 1);
        Serial.printf("Contrast %s\n", g_apply_contrast ? "ON" : "OFF");
    }
    else if (cmdUpper.startsWith("STRENGTH"))
    {
        int val = cmdUpper.substring(9).toInt();
        if (val >= 0 && val <= 100)
        {
            g_contrast_strength = val;
            Serial.printf("Contrast strength: %d\n", val);
        }
    }
    else if (cmdUpper.startsWith("MM"))
    {
        float mm = cmd.substring(3).toFloat();
        if (mm > 0)
        {
            g_mm_per_px = mm;
            g_use_mm = true;
            Serial.printf("Scale: %.3f mm/px for resolution %s\n",
                          mm, res_names[current_res_index]);
        }
    }
    else if (cmdUpper.startsWith("MINAREA"))
    {
        int v = cmdUpper.substring(8).toInt();
        if (v > 0)
        {
            g_min_area = v;
            Serial.printf("Min area: %d px\n", v);
        }
        else
            Serial.println("Usage: MINAREA <pixels>  e.g. MINAREA 100");
    }
    else if (cmdUpper.startsWith("MAXAREA"))
    {
        int v = cmdUpper.substring(8).toInt();
        if (v > 0)
        {
            g_max_area = v;
            Serial.printf("Max area: %d px\n", v);
        }
        else
            Serial.println("Usage: MAXAREA <pixels>  e.g. MAXAREA 200000");
    }
    else if (cmdUpper == "DIAGNOSE")
    {
        // Capture a single frame and print pixel statistics WITHOUT measuring
        // Helps figure out why detection is failing
        Serial.println("\n--- DIAGNOSE MODE ---");
        Serial.println("Capturing frame and computing image statistics...");

        sensor_t *s = esp_camera_sensor_get();
        if (s && g_exposure_locked)
        {
            s->set_exposure_ctrl(s, 0);
            s->set_aec_value(s, g_locked_aec_value);
            s->set_gain_ctrl(s, 0);
            s->set_agc_gain(s, g_locked_agc_gain);
        }
        // Warm up
        for (int i = 0; i < 5; i++)
        {
            camera_fb_t *tmp = esp_camera_fb_get();
            if (tmp)
                esp_camera_fb_return(tmp);
            delay(60);
        }
        camera_fb_t *fb = esp_camera_fb_get();
        if (!fb)
        {
            Serial.println("Capture failed");
            return;
        }

        int w = fb->width, h = fb->height;
        uint8_t *img = fb->buf;

        // Compute image histogram buckets
        uint32_t hist[8] = {0};
        uint8_t mn = 255, mx = 0;
        uint32_t sm = 0;
        for (int i = 0; i < w * h; i++)
        {
            uint8_t v = img[i];
            if (v < mn)
                mn = v;
            if (v > mx)
                mx = v;
            sm += v;
            hist[v / 32]++;
        }
        uint8_t mean = sm / (w * h);

        Serial.printf("Image size:  %dx%d = %d pixels\n", w, h, w * h);
        Serial.printf("Pixel range: %d - %d\n", mn, mx);
        Serial.printf("Mean:        %d\n", mean);
        Serial.println("Histogram (32-wide buckets):");
        for (int b = 0; b < 8; b++)
        {
            float pct = 100.0f * hist[b] / (w * h);
            Serial.printf("  %3d-%3d: %6lu px  (%.1f%%)\n",
                          b * 32, b * 32 + 31, hist[b], pct);
        }

        if (g_calibrated)
        {
            Serial.println("\nCalibration thresholds (center cell):");
            int gy = CALIB_GRID / 2, gx = CALIB_GRID / 2;
            float T = g_bg_mean[gy][gx] - g_k_sigma * g_bg_stddev[gy][gx];
            Serial.printf("  mean=%d, std=%d, k=%.1f => T=%.1f\n",
                          g_bg_mean[gy][gx], g_bg_stddev[gy][gx], g_k_sigma, T);
            Serial.printf("  Object pixels should be BELOW %.0f\n", T);
            Serial.printf("  Your image min=%d, mean=%d\n", mn, mean);
            if (mn > T)
                Serial.println("  !! Image minimum > threshold: object may be too bright to detect!");
            else
                Serial.printf("  OK: %d pixels below threshold\n",
                              [&]()
                              {int c=0; for(int i=0;i<w*h;i++) if(img[i]<(uint8_t)T) c++; return c; }());
        }
        else
        {
            uint8_t T = mean - g_thresh_offset;
            Serial.printf("\nNot calibrated. Fallback T=%d (mean %d - offset %d)\n",
                          T, mean, g_thresh_offset);
            int cnt = 0;
            for (int i = 0; i < w * h; i++)
                if (img[i] < T)
                    cnt++;
            Serial.printf("Pixels below T: %d (%.1f%%)\n", cnt, 100.0f * cnt / (w * h));
        }

        esp_camera_fb_return(fb);
        Serial.println("--- DIAGNOSE Complete ---\n");
        Serial.println("If 0 pixels below threshold: CALIBRATE or lower KSIGMA or raise OFFSET.");
        Serial.println("If too many pixels: raise KSIGMA or RECALIBRATE with better background.");
    }
    else
    {
        int count = cmdUpper.toInt();
        if (count > 0)
        {
            Serial.printf("Batch: %d images...\n", count);
            for (int i = 0; i < count; i++)
            {
                Serial.printf("\n[%d/%d]\n", i + 1, count);
                captureAndMeasure();
                delay(500);
            }
        }
        else
            Serial.println("Unknown command. Type HELP.");
    }
}

// =======================================================
// SETUP & LOOP
// =======================================================
void setup()
{
    Serial.begin(115200);
    delay(2000);

    WRITE_PERI_REG(RTC_CNTL_BROWN_OUT_REG, 0);

    Serial.println("\nESP32-CAM Measurement System Booting...");

    pinMode(FLASH_LED_PIN, OUTPUT);
    digitalWrite(FLASH_LED_PIN, LOW);

    if (!psramFound())
        Serial.println("WARNING: PSRAM not found - resolution limited");

    if (!initCamera())
    {
        Serial.println("FATAL: Camera init failed!");
        while (true)
            delay(1000);
    }

    initSD();
    printSystemInfo();
}

void loop()
{
    if (!Serial.available())
        return;
    String cmd = Serial.readStringUntil('\n');
    cmd.trim();
    if (cmd.length() > 0)
        processCommand(cmd);
}
